59 int computeFeaturePoseJacobian(vpMatrix &L, unsigned int nbDOF, bool *dof)

61 unsigned int numDOF;
62 unsigned long nbSamples = sampler.getNbSamples();
63 vpRowVector Ls;
64 double *pt_L, *pt_Ls;

77 it_set->getPoseJacobian(Ls);

80 pt_Ls = Ls.data;
