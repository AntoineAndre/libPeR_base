102void prCameraModel::meterPixelConversion(prPointFeature & P)
103{
104 if(distorsions)
105 {
106 double x = P.get_x(), y = P.get_y(), r2, distR, distR_den, xy, xd, yd;
107 r2 = x*x+y*y;
108 distR = 1;
109 xy = x*y;
110 if(activek[0])
111 distR += k[0]*r2;
112 if(activek[1])
113 distR += k[1]*r2*r2; //r4
114 if(activek[2])
115 distR += k[2]*r2*r2*r2; //r6
116
117 xd = x*distR;
118 yd = y*distR;
119
120 distR_den = 1;
121
122 if(activek[5])
123 distR_den += k[5]*r2;
124 if(activek[6])
125 distR_den += k[6]*r2*r2; //r4
126 if(activek[7])
127 distR_den += k[7]*r2*r2*r2; //r6
128
129 xd /= distR_den;
130 yd /= distR_den;
131
132 if(activek[3])
133 {
134 xd += 2*k[3]*xy;
135 yd += k[3]*(r2 + 2*y*y);
136 }
137 if(activek[4])
138 {
139 xd += k[4]*(r2 + 2*x*x);
140 yd += 2*k[4]*xy;
141 }
142
143 P.set_u(xd * au + u0);
144 P.set_v(yd * av + v0);
145 }
146 else
147 {
148 P.set_u(P.get_x() * au + u0);
149 P.set_v(P.get_y() * av + v0);

183

186
