76void prEquirectangular::computeSensorJacobian(prPointFeature & P, vpMatrix & LuX)
77{
78 double X = P.get_X(), Y = P.get_Y(), Z = P.get_Z();
79 double X2=X*X, Y2=Y*Y, Z2 = Z*Z;
80 double X2pZ2 = X2+Z2;
81
82 if(X2pZ2 < 1e-8)
83 {
84 LuX.resize(2,3,true);
85 }
86 else
87 {
88 //get du / dx from mother class
89 vpMatrix Lux(2,2);
90 prCameraModel::computeSensorJacobian(P, Lux);
91
92 double srX2pZ2 = sqrt(X2pZ2);
93 double X2pY2pZ2 = X2pZ2+Y2;
94
95 vpMatrix LxX(2,3);
96 LxX[0][0] = Z/X2pZ2; LxX[0][1] = 0.; LxX[0][2] = -X/X2pZ2;
97 LxX[1][0] = -X*Y/(srX2pZ2*X2pY2pZ2); LxX[1][1] = srX2pZ2/X2pY2pZ2; LxX[1][2] = -Y*Z/(srX2pZ2*X2pY2pZ2);
98
99 LuX = Lux*LxX;
100 }
